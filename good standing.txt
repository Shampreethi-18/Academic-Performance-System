import java.util.*;

class Student {
    int id;
    String name;
    double cgpa;
    String status;

    Student(int id, String name, double cgpa) {
        this.id = id;
        this.name = name;
        this.cgpa = cgpa;
        this.status = "Not Evaluated";
    }

    void display() {
        System.out.println("ID: " + id + " | Name: " + name +
                " | CGPA: " + cgpa +
                " | Status: " + status);
    }
}

public class GoodStandingSystem {
    static ArrayList<Student> students = new ArrayList<>();
    static Scanner sc = new Scanner(System.in);
    static final double CGPA_THRESHOLD = 5.0;

    public static void main(String[] args) {
        int choice;

        do {
            System.out.println("\n--- Good Standing System ---");
            System.out.println("1. Add Student");
            System.out.println("2. View Students");
            System.out.println("3. Assign Good Standing");
            System.out.println("4. Exit");
            System.out.print("Enter choice: ");

            choice = sc.nextInt();

            switch (choice) {
                case 1:
                    addStudent();
                    break;
                case 2:
                    viewStudents();
                    break;
                case 3:
                    assignGoodStanding();
                    break;
                case 4:
                    System.out.println("Exiting...");
                    break;
                default:
                    System.out.println("Invalid choice!");
            }
        } while (choice != 4);
    }

    static void addStudent() {
        System.out.print("Enter ID: ");
        int id = sc.nextInt();
        sc.nextLine();

        System.out.print("Enter Name: ");
        String name = sc.nextLine();

        System.out.print("Enter CGPA: ");
        double cgpa = sc.nextDouble();

        students.add(new Student(id, name, cgpa));
        System.out.println("Student added!");
    }

    static void viewStudents() {
        for (Student s : students) {
            s.display();
        }
    }

    static void assignGoodStanding() {
        for (Student s : students) {
            if (s.cgpa >= CGPA_THRESHOLD) {
                s.status = "Good Standing";
            }
        }
        System.out.println("Good Standing assigned to eligible students.");
    }
}
